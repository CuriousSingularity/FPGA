83/**************************************************************************



84 * resetIPCore



85 * Return: void



86 * Arguments: ipcore name



87 * Function: This function writes 0x0 to 0x4 offset register and 0x1 to 0x0



88 *           register to disable and reset the IP core.



89  *************************************************************************/



90void resetIPCore(const char * ipcore_name)  {



91    uint32_t a_val = 0x1;



92    uint32_t z_val = 0x0;



93    



94    AXI4_struct * this_state = axi4LiteInit(ipcore_name);



95    MW_DBG_printf("Reset IP Core - %s\n",this_state->ipcore_name);



96    /* write_0x0_to_0x4-Enable_register - Disable PCORE



97     * write_0x1_to_0x0-Reset_register  - Reset PCORE



98     * axi4LiteWrite(p_state, src_ptr, reg_off , datum_len)



99     * axi4LiteWrite(p_state, src_ptr, reg_off , datum_len)



100     */



101    axi4LiteWrite(this_state, &z_val, 0x4, 0x4);



102    axi4LiteWrite(this_state, &a_val, 0x0, 0x4);



103}

106AXI4_struct* axi4LiteInit(const char * ipcore_name)



107{



108    int oflags, i;



109    AXI4_struct* p_state = NULL;



110    AXI4_struct* p_state_list = p_axi4Lite_state_list;



111    size_t len;



112    



113    MW_DBG_functionIn();



114



115    pthread_once(&state_list_once, state_list_init);



116



117    if (ipcore_name == NULL) /* Check PCORE string */



118    {



119        fprintf(stderr,"ERROR: ipcore_name ptr is NULL.\n");



120        return NULL;



121    }



122    



123    pthread_mutex_lock(&state_list_mutex);



124    while(p_state_list!=NULL)



125    {



126        if(strcmp(ipcore_name,p_state_list->ipcore_name)==0)



127        {



128            p_state = p_state_list;



129            p_state->opened++;



130            MW_DBG_printf("Found existing open device: %s\n", p_state->ipcore_name);



131            pthread_mutex_unlock(&state_list_mutex);



132			MW_DBG_functionOut();



133            return p_state;



134        }



135        p_state_list = p_state_list->next;



136    }



137



138    /* No existing open device found */



139    p_state=(AXI4_struct*)malloc(sizeof(AXI4_struct));



140    len = strlen(ipcore_name);



141    p_state->ipcore_name = malloc(sizeof(char[len+1]));



142    strcpy(p_state->ipcore_name,ipcore_name);



143    p_state->prev = NULL;



144    p_state->next = NULL;



145    p_state->opened = 0;



146    



147    /* Open the /dev/pcore_name*/



148    p_state->fd = open(p_state->ipcore_name, O_RDWR | O_SYNC);



149    /* return NULL if file-descriptor was invalid */



150    if(p_state->fd < 0)



151    {



152        fprintf(stderr, "ERROR: Cannot open device: %s\n", p_state->ipcore_name);



153        perror("DEV_OPEN_FAILURE");



154



155        free(p_state->ipcore_name);



156        free(p_state);



157        pthread_mutex_unlock(&state_list_mutex);



158        return NULL;



159    }



160    



161    /* Map the base address to virtual address */



162    p_state->virtual_addr = (unsigned char*) mmap(NULL, MAP_LENGTH, PROT_READ | PROT_WRITE, MAP_SHARED, p_state->fd, 0);



163



164    /* When MMAP fails it returns MAP_FAILED (void*(-1)).



165     * In that case, print error and return NULL.



166     */



167    if (p_state->virtual_addr == MAP_FAILED) {



168        close(p_state->fd);



169        fprintf(stderr, "ERROR: MMAP Failed with device: %s\n", p_state->ipcore_name);



170        perror("MMAP_FAILURE");



171        free(p_state->ipcore_name);



172        free(p_state);



173        pthread_mutex_unlock(&state_list_mutex);



174        return NULL;



175    }



176



177    pthread_mutex_init(&p_state->mutex, NULL);



178    p_state->opened = 1;



179



180    if(p_axi4Lite_state_list!=NULL)



181    {



182        p_axi4Lite_state_list->prev = p_state;



183        p_state->next = p_axi4Lite_state_list;



184    }



185    p_axi4Lite_state_list = p_state;



186    pthread_mutex_unlock(&state_list_mutex);



187



188    MW_DBG_printf("Open fd success with device: %s\n", p_state->ipcore_name);



189	MW_DBG_functionOut(); 



190



191    return p_state;



192}

206void axi4LiteTerminate(AXI4_struct* p_state)



207{



208    



209    MW_DBG_functionIn();



210    if (p_state==NULL)



211    {



212        fprintf(stderr,"ERROR: AXI4_struct ptr is NULL.\n");



213        return;



214    }



215    



216    pthread_mutex_lock(&state_list_mutex);



217   



218    p_state->opened--;



219



220    if(p_state->opened > 0)



221    {



222        MW_DBG_printf("Not removing device %s; %d remaining instance\n", p_state->ipcore_name,p_state->opened);



223        pthread_mutex_unlock(&state_list_mutex);



224		MW_DBG_functionOut();



225        return;



226    }



227    



228    MW_DBG_printf("Removing device %s\n", p_state->ipcore_name);



229    if(p_axi4Lite_state_list == p_state)



230    {



231        p_axi4Lite_state_list = p_state->next;



232    }



233    if(p_state->prev != NULL)



234    {



235        p_state->prev->next = p_state->next;



236    }



237    if(p_state->next != NULL)



238    {



239        p_state->next->prev = p_state->prev;



240    }



241



242    pthread_mutex_destroy(&p_state->mutex);



243



244    MW_DBG_printf("Device %s Un-mapping %d bytes of %#x\n", p_state->ipcore_name, MAP_LENGTH, (unsigned int)p_state->virtual_addr);



245    munmap((&p_state->virtual_addr), MAP_LENGTH);



246    close(p_state->fd);



247    free(p_state->ipcore_name);



248    free(p_state);



249    p_state = NULL;



250



251    pthread_mutex_unlock(&state_list_mutex);



252



253    MW_DBG_functionOut();



254    return;



255}

257/**************************************************************************



258 * axi4LiteRead



259 * Return: void



260 * Arguments: offset, data length, dest-pointer, data type



261 * Function: This wrapper function calculates the length of data in bytes



262 *           uses the length calculated in bytes to copy the data to dest



263 *           void pointer.



264 *************************************************************************/



265int axi4LiteRead_deprecated(unsigned int offset, unsigned int numData, void * dest, unsigned int dataType)



266{



267    unsigned int dataLength = DType[dataType] * numData;



268    /*



269     * For fixed point data-types fixdt(1,32,*) assume dataType bytes = 4



270     */



271    MW_DBG_printf("Read data type is : %u and data Length: %u\n", dataType,numData);



272    if (dataType > 9) {



273        dataLength = sizeof(unsigned long)*numData;



274    }



275    



276    return axi4LiteRead(gp_state_deprecated, dest, offset, dataLength);



277}

279int axi4LiteRead(AXI4_struct* p_state, void * dest, unsigned int offset, unsigned int dataLength)



280{



281	void* src;



282	unsigned int i;



283



284    MW_DBG_functionIn();



285



286    if (p_state==NULL)



287    {



288        fprintf(stderr,"ERROR: AXI4_struct ptr is NULL.\n");



289        return -1;



290    }



291



292#ifdef _DEBUG



293    if ( (p_state->fd < 0) || (p_state->virtual_addr == MAP_FAILED) )



294    {



295        fprintf(stderr, "ERROR: Invalid file descriptor or MAP_FAILED:fd - %d, virtualAddr: %#lx with device %s.\n", p_state->ipcore_name, p_state->fd, p_state->virtual_addr);



296        return -1;



297    }



298#endif



299



300    //pthread_mutex_lock(&p_state->mutex);   



301



302    MW_DBG_printf("Device %s read data and data Length: %u\n", p_state->ipcore_name,dataLength);



303    MW_DBG_printf("Device %s AXIInterface: AXI4Lite Register offset: %#lx, Num Bytes : %lu \n", p_state->ipcore_name, offset, dataLength);



304



305	src = (void*) (p_state->virtual_addr + offset);



306



307	if(!(((uintptr_t)src) & 0x3) && !(dataLength & 0x3))



308	{



309		uint32_t* dest32 = (uint32_t*) dest;



310		const uint32_t volatile* src32 = (uint32_t*) src;



311



312        for (i=0;i<dataLength;i+=4)



313		{



314            *dest32++ = *src32++;



315		}



316	}



317	else if(!(((uintptr_t)src) & 0x1) && !(dataLength & 0x1))



318	{



319		uint16_t* dest16 = (uint16_t*) dest;



320		const uint16_t volatile* src16 = (uint16_t*) src;



321



322        for (i=0;i<dataLength;i+=2)



323		{



324            *dest16++ = *src16++;



325		}



326	}



327	else



328	{



329		uint8_t* dest8 = (uint8_t*) dest;



330		const uint8_t volatile* src8 = (uint8_t*) src;



331



332        for (i=0;i<dataLength;i++)



333		{



334            *dest8++ = *src8++;



335		}



336	}



337



338	//pthread_mutex_unlock(&p_state->mutex);



339



340    return 0;



341}
